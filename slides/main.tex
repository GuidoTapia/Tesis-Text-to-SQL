\begin{frame}{Pipeline propuesto}
\begin{figure}
    \centering
    \includegraphics[width=0.96\textwidth, keepaspectratio]{graficos/pipeline-general.png}
\end{figure}
\vspace{-0.2cm}
\begin{itemize}\footnotesize
    \item \textbf{(1) Anclaje al esquema dual.} Construye $\Sigma_{\mathrm{proj}} = \langle T_{\mathrm{rel}}, V_{\mathrm{grafo}}, E_{\mathrm{grafo}}, C_{\mathrm{val}} \rangle$.
    \item \textbf{(2) Interpretación estructurada} de la pregunta.
    \item \textbf{(3) Construcción de la RI tipada} IR-SQL/PGQ.
    \item \textbf{(4) Compilación determinista} a SQL/PGQ.
    \item \textbf{(5) Verificación} estática y dinámica con bucle de refinamiento sobre la RI.
\end{itemize}
\end{frame}

\vspace{0.3cm}
\begin{block}{Procedencia y estado}
\small \textbf{M-Schema} de XiYan-SQL~\parencite{XiYanSQL2024} extendido con descriptores de grafo, \textbf{enlazado bidireccional} de RSL-SQL~\parencite{RSLSQL2024} y \textbf{recuperación} estilo CHESS~\parencite{CHESS2024}. Hoy opera (1) sobre la estructura completa; (2) y (3) se incorporarán para esquemas masivos como BIRD donde el filtrado por relevancia es indispensable.
\end{block}
\end{frame}

\begin{frame}{IR-SQL/PGQ: vista intuitiva}
\textbf{Ejemplo.} \emph{¿Qué empleados conocen a alguien del equipo de Marketing y trabajan en Lima?}


\centering
\begin{tikzpicture}[node distance=0.6cm]
    \node[tkbox, minimum width=3.4cm, minimum height=2.0cm] (nl) {\scriptsize \textbf{NL}\\\scriptsize\itshape Empleados que\\conocen a alguien\\de Marketing y\\trabajan en Lima};
    \node[tkboxo, minimum width=4.0cm, minimum height=2.0cm, right=of nl] (ir) {\scriptsize \textbf{IR-SQL/PGQ}\\\scriptsize\texttt{RelationalQuery}\\\scriptsize\texttt{ \textbar-- FromGraphMatch}\\\scriptsize\texttt{ \ \ \ MatchPattern}\\\scriptsize\texttt{ \ \ \ \ \ Person--$>$Person}\\\scriptsize\texttt{ \textbar-- Filter: city=Lima}};
    \node[tkboxg, minimum width=3.6cm, minimum height=2.0cm, right=of ir] (sql) {\scriptsize \textbf{SQL/PGQ}\\\scriptsize\texttt{SELECT e.name}\\\scriptsize\texttt{FROM GRAPH\_TABLE(}\\\scriptsize\texttt{ MATCH (a:Person)}\\\scriptsize\texttt{ -[k:knows]-$>$}\\\scriptsize\texttt{ (b:Person) ...)}};
    \draw[tkarrow] (nl) -- node[above, tklabel] {LLM} (ir);
    \draw[tkarrow] (ir) -- node[above, tklabel] {compilador} (sql);
\end{tikzpicture}
\vspace{0.2cm}
\begin{block}{Idea central}
La RI es el \textbf{contrato fijo} entre el modelo y el resto del pipeline. El LLM \emph{nunca} emite un query libre: emite IR validada por tipos, que un compilador determinista traduce a SQL/PGQ.
\end{block}
\end{frame}

\begin{frame}{IR-SQL/PGQ: árbol y tres bloques cerrados}
\centering
\begin{tikzpicture}[node distance=0.4cm and 0.6cm]
    \node[tkbox, minimum width=3.0cm] (rel) {\textbf{Bloque relacional}\\\scriptsize\texttt{TableRef}, \texttt{ColumnRef}\\\scriptsize Expresiones recursivas\\\scriptsize \texttt{RelationalQuery}};
    \node[tkboxg, minimum width=3.0cm, right=of rel] (graph) {\textbf{Bloque de grafo}\\\scriptsize\texttt{VertexPattern}\\\scriptsize\texttt{EdgePattern}, \texttt{PathPattern}\\\scriptsize\texttt{MatchPattern}};
    \node[tkboxo, minimum width=3.0cm, right=of graph] (hyb) {\textbf{Composición híbrida}\\\scriptsize\texttt{FromGraphMatch}\\\scriptsize Encaja en \texttt{FromExpr}\\\scriptsize Tres ámbitos separados};
    \node[font=\scriptsize\bfseries, below=0.4cm of graph] (link) {Cualquier consulta del alcance es composición de estos tres bloques};
    \draw[tkarrow, dashed] (graph) -- (link);
\end{tikzpicture}

\vspace{0.4cm}
\textbf{Linaje del diseño}
\begin{itemize}\footnotesize
    \item \textbf{SemQL}~\parencite{IRNet}: descomposición tipada.
    \item \textbf{NatSQL}: uniformidad operacional.
    \item \textbf{RESDSQL}~\parencite{RESDSQL2023}: compilador determinista.
\end{itemize}

\end{frame}

\textbf{Estado actual.}
\begin{itemize}\small
